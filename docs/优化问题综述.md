$$ \min_{x \in \Omega} \quad \lambda_1 C_{\text{sla}}(x) + \lambda_2 C_{\text{res}}(x) + \lambda_3 C_{\text{chg}}(x) +
  \lambda_4 C_{\text{risk}}(x)$$

---
  决策变量

  $$\begin{aligned}
  x_{f,s} &\in \{0,1\} \quad &&\text{流 } f \text{ 是否映射到切片 } s \\
  B_{f,s}^{\text{ul}}, B_{f,s}^{\text{dl}} &\in \mathbb{R}^+ \quad &&\text{流 } f \text{ 在切片 } s \text{
  上分配的上/下行带宽} \\
  \mathit{served}_f &\in \{0,1\} \quad &&\text{流 } f \text{ 是否被服务} \\
  \mathit{deficit}_f^{\text{ul}}, \mathit{deficit}_f^{\text{dl}} &\in \mathbb{R}^+ \quad &&\text{流 } f \text{
  的带宽缺口} \\
  \mathit{gbr\_gap}_f^{\text{ul}}, \mathit{gbr\_gap}_f^{\text{dl}} &\in \mathbb{R}^+ \quad &&\text{流 } f \text{ 的 GBR
  缺口} \\
  \mathit{dev}_s &\in \mathbb{R}^+ \quad &&\text{切片 } s \text{ 的负载偏离度} \\
  \mathit{change}_{f,s} &\in \mathbb{R}^+ \quad &&\text{流 } f \text{ 映射变更的线性化变量}
  \end{aligned}$$

---
  约束空间 $(\Omega)$

  $$\begin{aligned}
  \text{(C1) 流映射唯一性:} \quad & \sum_{s} x_{f,s} = \mathit{served}_f \leq 1, \quad \forall f \\[6pt]
  \text{(C2) 带宽平衡:} \quad & \sum_{s} B_{f,s}^{\text{ul}} + \mathit{deficit}_f^{\text{ul}} =
  \mathit{req}_f^{\text{ul}}, \quad \forall f \\
  & \sum_{s} B_{f,s}^{\text{dl}} + \mathit{deficit}_f^{\text{dl}} = \mathit{req}_f^{\text{dl}}, \quad \forall f \\[6pt]
  \text{(C3) GBR 下限:} \quad & \sum_{s} B_{f,s}^{\text{ul}} + \mathit{gbr\_gap}_f^{\text{ul}} \geq
  \mathit{gbr}_f^{\text{ul}}, \quad \forall f \\
  & \sum_{s} B_{f,s}^{\text{dl}} + \mathit{gbr\_gap}_f^{\text{dl}} \geq \mathit{gbr}_f^{\text{dl}}, \quad \forall f
  \\[6pt]
  \text{(C4) SLA 硬过滤:} \quad & x_{f,s} = 0 \quad \text{if} \quad \mathit{lat}_s + \mathit{proc}_s >
  \mathit{lat}_f^{\max} \\
  & x_{f,s} = 0 \quad \text{if} \quad \mathit{loss}_s > \mathit{loss}_f^{\max} \\
  & x_{f,s} = 0 \quad \text{if} \quad \mathit{jitter}_s > \mathit{jitter}_f^{\max} \\[6pt]
  \text{(C5) 带宽上界:} \quad & B_{f,s}^{\text{ul}} \leq x_{f,s} \cdot \mathit{req}_f^{\text{ul}}, \quad \forall f,s \\
  & B_{f,s}^{\text{dl}} \leq x_{f,s} \cdot \mathit{req}_f^{\text{dl}}, \quad \forall f,s \\[6pt]
  \text{(C6) 切片容量:} \quad & \sum_{f} B_{f,s}^{\text{ul}} \leq \mathit{Cap}_s^{\text{ul}} -
  \mathit{bg}_s^{\text{ul}}, \quad \forall s \\
  & \sum_{f} B_{f,s}^{\text{dl}} \leq \mathit{Cap}_s^{\text{dl}} - \mathit{bg}_s^{\text{dl}}, \quad \forall s \\[6pt]
  \text{(C7) 负载均衡:} \quad & \mathit{dev}_s \geq \left| \frac{\sum_f B_{f,s}^{\text{dl}}}{\mathit{Cap}_s^{\text{dl}}}\rho_s \right|, \quad \forall s \\
& \mathit{dev}_s \geq \left| \frac{\sum_f B_{f,s}^{\text{ul}}}{\mathit{Cap}_s^{\text{ul}}} - \rho_s \right|, \quad
  \forall s \\[6pt]
  \text{(C8) 变更线性化:} \quad & \mathit{change}_{f,s} \geq x_{f,s} - x_{f,s}^{\text{old}}, \quad \forall f,s \\
  & \mathit{change}_{f,s} \geq x_{f,s}^{\text{old}} - x_{f,s}, \quad \forall f,s \\[6pt]
  \text{(C9) 节点资源:} \quad & \text{CPU}^{\text{old}} + \sum_f B_f \cdot \alpha \leq \text{CPU}^{\max} \\
  & \text{PRB}^{\text{old}} + \sum_f B_f \cdot \beta \leq \text{PRB}^{\max}
  \end{aligned}$$

---
  目标函数各项
$$
\begin{aligned}
  C_{\text{sla}}(x) &= \frac{1}{|F|}\sum_{f} \frac{1}{\mathit{prio}_f} \Bigg[
      (1 - \mathit{served}_f) \\
      &\quad + \frac{\mathit{deficit}_f^{\text{ul}}}{\max(\mathit{req}_f^{\text{ul}}, \varepsilon)}
        + \frac{\mathit{deficit}_f^{\text{dl}}}{\max(\mathit{req}_f^{\text{dl}}, \varepsilon)} \\[4pt]
      &\quad + 0.5 \cdot \frac{\mathit{gbr\_gap}_f^{\text{ul}}}{\max(\mathit{gbr}_f^{\text{ul}}, \varepsilon)}
        + 0.5 \cdot \frac{\mathit{gbr\_gap}_f^{\text{dl}}}{\max(\mathit{gbr}_f^{\text{dl}}, \varepsilon)} \\[4pt]
      &\quad + \sum_{s} \mathbb{I}(\mathit{lat}_s + \mathit{proc}_s > \mathit{lat}_f^{\max}) \cdot x_{f,s} \\
      &\quad + \sum_{s} \mathbb{I}(\mathit{loss}_s > \mathit{loss}_f^{\max}) \cdot x_{f,s}
        + \sum_{s} \mathbb{I}(\mathit{jitter}_s > \mathit{jitter}_f^{\max}) \cdot x_{f,s}
  \Bigg] \\[16pt]

  C_{\text{res}}(x) &= \frac{1}{|S|}\sum_{s} \mathit{dev}_s \\[12pt]

  C_{\text{chg}}(x) &= \frac{1}{|F|}\sum_{f,s} \mathit{change}_{f,s} \\[12pt]

  C_{\text{risk}}(x) &= \gamma_{\text{mobility}} \cdot C_{\text{chg}}^{\text{am}}(x)
  \end{aligned}
$$


---
  权重映射
$$


  \begin{aligned}
  \lambda_1 &= 800 \cdot \omega_{\text{sla}} + 200 \cdot \omega_{\text{fairness}} \quad &&(\text{默认 } 860) \\
  \lambda_2 &= 100 \cdot \omega_{\text{resource}} \quad &&(\text{默认 } 70) \\
  \lambda_3 &= 50 \cdot \omega_{\text{churn}} \quad &&(\text{默认 } 25) \\
  \lambda_4 &= 20 \cdot \omega_{\text{risk}} \quad &&(\text{默认 } 16)
  \end{aligned}
$$
  其中语义权重 (\omega) 由 ObjectiveProfile 按运行模板设定：

$$
  \begin{array}{l|cccc|c}
  \text{Profile} & \omega_{\text{sla}} & \omega_{\text{resource}} & \omega_{\text{churn}} & \omega_{\text{risk}} &
  \text{适用场景} \\
  \hline
  \text{balanced}        & 1.0 & 0.7 & 0.5 & 0.8 & \text{日常运行} \\
  \text{QOS\_FIRST}      & \geq 1.4 & \geq 0.8 & \leq 0.4 & \leq 0.5 & \text{SLA 优先} \\
  \text{STABILITY\_FIRST} & \leq 0.9 & \leq 0.5 & \geq 1.2 & - & \text{变更窗口外} \\
  \text{CONGESTION\_RELIEF}& \geq 1.0 & \geq 1.3 & \leq 0.35 & - & \text{切片拥塞} \\
  \text{MOBILITY\_FIRST}  & \geq 0.9 & - & \geq 0.7 & \geq 1.4 & \text{高速移动} \\
  \end{array}
$$


---
  符号表
$$
  \begin{aligned}
  F     &\quad \text{流集合} \\
  S     &\quad \text{切片集合} \\
  \mathit{prio}_f &\quad \text{流 } f \text{ 的 SLA 优先级} \\
  \mathit{req}_f^{\text{ul}}, \mathit{req}_f^{\text{dl}} &\quad \text{流 } f \text{ 的上/下行请求带宽} \\
  \mathit{gbr}_f^{\text{ul}}, \mathit{gbr}_f^{\text{dl}} &\quad \text{流 } f \text{ 的上/下行保障带宽 (GBR)} \\
  \mathit{lat}_f^{\max}, \mathit{loss}_f^{\max}, \mathit{jitter}_f^{\max} &\quad \text{流 } f \text{ 的 SLA 门限} \\
  \mathit{lat}_s, \mathit{loss}_s, \mathit{jitter}_s, \mathit{proc}_s &\quad \text{切片 } s \text{ 的 QoS 属性} \\
  \mathit{Cap}_s^{\text{ul}}, \mathit{Cap}_s^{\text{dl}} &\quad \text{切片 } s \text{ 的容量} \\
  \mathit{bg}_s^{\text{ul}}, \mathit{bg}_s^{\text{dl}} &\quad \text{切片 } s \text{ 的背景负载} \\
  \rho_s     &\quad \text{切片 } s \text{ 的目标负载比} \\
  \gamma_{\text{mobility}} &\quad \text{移动性风险评分 } (0 \sim 1) \\
  \mathbb{I}(\cdot) &\quad \text{指示函数，条件成立取 } 1 \text{，否则取 } 0 \\
  \varepsilon &\quad \text{防除零小量}
  \end{aligned}
$$
